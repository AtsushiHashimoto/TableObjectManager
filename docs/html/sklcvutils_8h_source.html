169  template<class T> void estimate_noise(const cv::Mat& src, std::vector<T>& std_dev){
170  assert(src.type()==CV_8UC3);
171  size_t channels = src.channels();
172 
173  std::vector<std::vector<T> > abs_diff(channels),diff(channels);
174 
175  const cv::Vec3b* upper_column;
176  const cv::Vec3b* lower_column;
177  T temp;
178 
179  for(int y=0,idx=0;y<src.rows;y++){
180  upper_column = src.ptr<cv::Vec3b>(y);
181  if(y!=src.rows-1){
182  lower_column = src.ptr<cv::Vec3b>(y+1);
183  }
184  else{
185  lower_column = NULL;
186  }
187 
188  for(int x=0;x<src.cols;x++,idx++){
189  for(size_t c=0;c<channels;c++){
190  if(!isInRange((int)upper_column[x][c],0,255)) continue;
191 
192  if(x<src.cols-1){
193  if(!isInRange((int)upper_column[x+1][c],0,255)) continue;
194  temp = (T)upper_column[x][c] - (T)upper_column[x+1][c];
195 // if(temp==0) continue;
196  diff[c].push_back(temp);
197  abs_diff[c].push_back(std::abs(temp));
198  }
199 
200  if(lower_column!=NULL){
201  if(!isInRange((int)lower_column[x][c],0,255)) continue;
202  temp = (T)upper_column[x][c]-(T)lower_column[x][c];
203 // if(temp==0) continue;
204  diff[c].push_back(temp);
205  abs_diff[c].push_back(std::abs(temp));
206  }
207  }
208  }
209  }
210 
211  std_dev.resize(channels);
212  for(size_t c=0;c<channels;c++){
213  size_t valid_num = diff[c].size();
214  size_t median_index = valid_num/2;
215  assert(valid_num>0);
216  std::nth_element(abs_diff[c].begin(),abs_diff[c].begin()+median_index,abs_diff[c].end());
217 
218  T threshold = std::max(abs_diff[c][median_index],1.f) * static_cast<T>(4.f/MEDIAN_SD_RATIO);
219 // std::cerr << "std_dev by median [" << c << "] = " << (threshold/2) << std::endl;
220  T mean(0), square_mean(0);
221  abs_diff[c].clear();
222  size_t count(0);
223  for(size_t i=0;i<diff[c].size();i++){
224  if(std::abs(diff[c][i]) > threshold) continue;
225  mean += diff[c][i];
226  square_mean += std::pow(diff[c][i],static_cast<T>(2));
227  count++;
228  }
229  square_mean /= count;
230  mean /= count;
231  std_dev[c] = std::sqrt(square_mean - std::pow(mean,static_cast<T>(2)));
232 // std::cerr << "std_dev[" << c << "] = " << std_dev[c] << std::endl;
233  }
